[docs]def load_json(name):
    full_path = get_full_path(name)
    try:
        with open(full_path, 'r') as j:
            return json.load(j)
    except FileNotFoundError:
        print('File: "{}" not found'.format(full_path))
        raise FileNotFoundError

        json.dump(obj, j, indent=6)
